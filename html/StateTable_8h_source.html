25// Comment from cisst library
26// The state data table is the storage for the state of the task that
27// the table is associated with. It is a heterogenous circular buffer
28// and can contain data of any type so long as it is derived from
29// mtsGenericObject. The state data table also resolves conflicts
30// between reads and writes to the state, by ensuring that the reader
31// head is always one behind the write head. To ensure this we have an
32// assumption here that there is only one writer, though there can be
33// multiple readers. State Data Table is also refered as Data Table or
34// State Table elsewhere in the documentation.

393// CKim - Add new element to the State Table and returns the id of the column
394// Adding a new column (whose length is HistoryLength, name is 'name' and stores '_elementType' data) to the table,
395// The pointer to this column is stored in 'StateVector'. 'Name' is stored in 'StateVectorDataNames'
396// 'Accessor' to read data of each column is stored in 'StateVectorAccessors' and
397// Pointer to the data that will be recorded to each column are casted inton std::any and stored in
398// 'StateVectorElements'
399template <class _elementType>
400StateDataId StateTable::NewElement(const std::string & name, _elementType* pElement) {
401// typedef typename mtsGenericTypes<_elementType>::FinalType FinalType;
402// typedef typename mtsGenericTypes<_elementType>::FinalRefType FinalRefType;
403// mtsStateArray<FinalType> * elementHistory =
404// new mtsStateArray<FinalType>(*element, HistoryLength);
405
406 // CKim - 'StateVectorDataName' stores the name of the column
407 StateVectorDataNames.push_back(name);
408
409 // CKim - Create a column vector storing '_elementType' data and 'HistoryLength' long
410 StateArray<_elementType> * elementHistory = new StateArray<_elementType>(*pElement, HistoryLength);
411 StateVector.push_back(elementHistory);
412
413 // CKim - Assign Id of the column
414 StateDataId id = static_cast<StateDataId>(StateVector.size() - 1);
415
416 // CKim - 'StateVectorAccessors' store the 'Accessor' of the added column
417 AccessorBase * accessor = new Accessor<_elementType>(*this, id, elementHistory, pElement);
418 StateVectorAccessors.push_back(accessor);
419
420 // CKim - 'StateVectorElement' is a row of the StateTable, which will
421 // consists of pointers to different data type, and the data will be
422 // written to the table when 'Advance' is called.
423 // Original implementation in CISST library used vector of 'mtsGenericObject*',
424 // in this implementation, we will use std::any introduced in C++17
425 // set(CMAKE_CXX_STANDARD 17) / set(CMAKE_CXX_STANDARD_REQUIRED ON)
426 // in CMake to include C++ 17 lbraries
427 // https://en.cppreference.com/w/cpp/utility/any/type
428 // FinalRefType *pdata = mtsGenericTypes<_elementType>::ConditionalWrap(*element);
429 // StateVectorElements.push_back(pdata);
430 // Pointer to the state variable will be converted to std::any
431 StateVectorElements.push_back(pElement);
432
433
434 return id;
435}
